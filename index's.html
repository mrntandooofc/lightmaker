function uploadProject() { const fileInput = document.getElementById('project-file'); const file = fileInput.files[0]; if (file) { const formData = new FormData(); formData.append('file', file); fetch('/upload', { method: 'POST', body: formData }) .then(response => response.json()) .then(data => console.log(data)) .catch(error => console.error(error)); } else { console.log('No file selected'); } }
